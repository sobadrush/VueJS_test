Lo-Dash 測試

fp.filter測試

var arr = [ { 'name' : 'Roger' , 'age' : 18 } , { 'name' : 'Kelly' , 'age' : 50 } , { 'name' : 'Linda' , 'age' : 23 } , { 'name' : 'Linda' , 'age' : 45 } , ];
// 如果只是要回傳某個值，可以省略 return
fp.filter( (x) => x.age > 30 )(arr);
// 回傳 { 'name' : 'Kelly' , 'age' : 50 } { 'name' : 'Linda' , 'age' : 45 }
// 回傳 { 'name' : 'Linda' , 'age' : 23 }
fp.filter( (x) => {
   return x.name.startsWith('L') && x['age'] < 25;
})(arr);

fp.remove測試

var arr = [ { 'name' : 'Roger' , 'age' : 18 } , { 'name' : 'Kelly' , 'age' : 50 } , { 'name' : 'Linda' , 'age' : 23 } , { 'name' : 'Linda' , 'age' : 45 } , ];
// 如果只是要回傳某個值，可以省略 return
fp.remove( (x) => x.age > 30 )(arr);
// 回傳 { 'name' : 'Roger' , 'age' : 18 } { 'name' : 'Linda' , 'age' : 23 }
// 移除 age < 25 且 姓名由 'L'打頭的
// 回傳：
// { 'name' : 'Roger' , 'age' : 18 }
// { 'name' : 'Kelly' , 'age' : 50 }
// { 'name' : 'Linda' , 'age' : 45 }
fp.remove( (x) => {
   return x.name.startsWith('L') && x['age'] < 25;
})(arr);

fp.map測試

var arr = [ { 'name' : 'Roger' , 'age' : 18 } , { 'name' : 'Kelly' , 'age' : 50 } , { 'name' : 'Linda' , 'age' : 23 } , { 'name' : 'Linda' , 'age' : 45 } , ];
fp.map( x => x.age )(arr)
// 回傳 [18, 50, 23, 45]
fp.map( x => x.name )(arr)
// 回傳 ["Roger", "Kelly", "Linda", "Linda"]
fp.map( x => x.name + '@@@' )(arr)
// 回傳 ["Roger@@@", "Kelly@@@", "Linda@@@", "Linda@@@"]
